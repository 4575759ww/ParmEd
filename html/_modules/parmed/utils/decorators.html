    HAS_OPENMM = False


[docs]def needs_openmm(fcn):
    global HAS_OPENMM
    @wraps(fcn)
    def new_fcn(*args, **kwargs):
        if not HAS_OPENMM:
            raise ImportError('Could not find or import OpenMM')
        return fcn(*args, **kwargs)

    return new_fcn
